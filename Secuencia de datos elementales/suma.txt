CONSIGNA: dado dos numeros enteros ingresado por el usuario, mostrar la suma de ambos.


ACCION suma_ES
	
	AMBIENTE

	 VAR

     	num1 , num2 , result : ENTERO;



     	ALGORITMO

     		ESCRIBIR("Por favor ingrese el primer numero a sumar:");
     		LEER(num1);

			ESCRIBIR("Por favor ingrese el segundo numero a sumar:");
     		LEER(num2);     		   

     		result := num1 + num2;

     		ESCRIBIR("La suma de ambos numeros es: ", result);

     	 FIN ALGORITMO


 FIN ACCION
