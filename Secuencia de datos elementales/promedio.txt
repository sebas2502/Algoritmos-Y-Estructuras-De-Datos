CONSIGNA: dada tres notas de parciales ingresadas por el usuario, calcular el promedio e informar si el usuario o alumno aprobó la materia.
          (se aprueba con nota mayor o igual a 6)


ACCION suma_ES
	
	AMBIENTE

	 VAR

     	nota1 , nota2 , nota3 , prom : REAL;



     	ALGORITMO

     		ESCRIBIR("Por favor ingrese la nota del primer parcial :");
     		LEER(nota1);

			ESCRIBIR("Por favor ingrese la nota del segundo parcial :");
     		LEER(nota2);     

               ESCRIBIR("Por favor ingrese la nota del tercer parcial :");
               LEER(nota3); 		   

     		prom := (nota1 + nota2 + nota3) / 3;

     		SI (prom >= 6) ENTONCES
                
                    ESCRIBIR("El alumno aprobó la materia con un promedio de : " , prom);
                
                SINO

                 ESCRIBIR("El alumno no aprobó la materia, su promedio es: " , prom);

                 FIN SI

     	 FIN ALGORITMO


 FIN ACCION
