Se dispone de una secuencia de caracteres y se desea obtener una secuencia de salida 
que resulte de copiar la secuencia de entrada, descartando el caracter "$".



ACCION EJ_2.1.3_Es
	
	AMBIENTE

		VAR 

		sec , sal : secuencia de caracteres;
		v : caracter;


		ALGORITMO

			ARR(sec); AVZ(sec , v); CREAR(sal);
																												
			MIENTRAS NFDS (sec) HACER						

				SI (v <> '$') ENTONCES
				 
					ESCRIBIR(sal , v);

				 FIN SI

			 	AVZ(sec , v);	

			 FIN MIENTRAS

			CERRAR(sec); CERRAR(sal);
 
		 FIN ALGORITMO

 FIN ACCION
